rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Helper functions
    function isAuthenticated() {
      return request.auth != null;
    }

    function isAdmin() {
      return isAuthenticated() && request.auth.token.email in [
        // Add admin email addresses here
        'admin@gargeesolar.in' // Replace with actual admin emails
      ];
    }

    // ===========================================
    // LEADS - Quick Lead Popup (Public Write, Admin Read)
    // ===========================================
    match /leads/{leadId} {
      // Allow anyone to submit quick lead forms
      allow create: if request.resource.data.keys().hasAny(['name', 'phone']) &&
        request.resource.data.name is string;
      // Allow read for development
      allow read, update, delete: if true;
    }

    // ===========================================
    // CONTACT MESSAGES - Contact Form (Public Write, Admin Read)
    // ===========================================
    match /contact_messages/{messageId} {
      // Allow anyone to submit contact forms
      allow create: if request.resource.data.keys().hasAny(['name', 'message']) &&
        request.resource.data.name is string;
      // Allow read for development
      allow read, update, delete: if true;
    }

    // Projects collection - Public read, Admin write
    match /projects/{projectId} {
      allow read: if true; // Public read access for website display
      allow write: if true; // Temporarily allow write for development
      allow update: if true; // Temporarily allow update for development
      allow delete: if true; // Temporarily allow delete for development
    }

    // Brands collection - Public read, Admin write
    match /brands/{brandId} {
      allow read: if true; // Public read access for website display
      allow write: if true; // Temporarily allow write for development
      allow update: if true; // Temporarily allow update for development
      allow delete: if true; // Temporarily allow delete for development
    }

    // Contact Messages (legacy camelCase name) - Keep for backward compatibility
    match /contactMessages/{messageId} {
      allow read: if true; // Temporarily allow read for development
      allow write: if true; // Allow anyone to submit contact forms
      allow update: if true; // Temporarily allow update for development
      allow delete: if true; // Temporarily allow delete for development
    }

    // Career Applications collection - Admin only
    match /careerApplications/{applicationId} {
      allow read: if true; // Temporarily allow read for development
      allow write: if true; // Allow anyone to submit applications
      allow update: if true; // Temporarily allow update for development
      allow delete: if true; // Temporarily allow delete for development
    }

    // Employees collection - Restricted access
    match /employees/{employeeId} {
      allow read: if true; // Temporarily allow read for development
      allow write: if true; // Temporarily allow write for development
      allow update: if true; // Temporarily allow update for development
      allow delete: if true; // Temporarily allow delete for development
    }

    // Customers collection - Admin and Employee access
    match /customers/{customerId} {
      allow read: if true; // Temporarily allow read for development
      allow write: if true; // Temporarily allow write for development
      allow update: if true; // Temporarily allow update for development
      allow delete: if true; // Temporarily allow delete for development
    }

    // Installations collection - Admin and Employee access
    match /installations/{installationId} {
      allow read: if true; // Temporarily allow read for development
      allow write: if true; // Temporarily allow write for development
      allow update: if true; // Temporarily allow update for development
      allow delete: if true; // Temporarily allow delete for development
    }
  }
}
